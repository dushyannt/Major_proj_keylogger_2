function getTripStatus(tour: { startDate: string; endDate: string; enrollment?: { status: string } }): TripStatus {
  // If enrollment is cancelled, it's a cancelled trip
  if (tour.enrollment?.status === "cancelled") {
    return "cancelled";
  }
  
  const now = Date.now();
  const startDate = new Date(tour.startDate).getTime();
  const endDate = new Date(tour.endDate).getTime();
  
  if (now < startDate) {
    return "upcoming";
  } else if (now >= startDate && now <= endDate) {
    return "ongoing";
  } else {
    return "past";
  }
}
